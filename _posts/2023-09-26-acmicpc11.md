# 1940 주몽

## vector

```c++
#include <iostream>
#include <vector>
using namespace std;

int arr[15004];
int n, m;
int cnt;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n >> m;

	for (int i = 0; i < n; i++)
		cin >> arr[i];
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			if (arr[i] + arr[j] == m)
			{
				cnt++;
				continue;
			}
		}
	}
	cout << cnt << "\n";

	return 0;
}
```
