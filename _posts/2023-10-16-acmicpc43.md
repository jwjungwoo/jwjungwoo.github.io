# 13144 List of Unique Numbers

## array cnt

```c++
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

typedef unsigned long long ull;
int n, l, r;
int cnt[100004];
ull ret;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n;
	vector<int> v(n);
	for (int i = 0; i < n; i++)
		cin >> v[i];
	
	while (r < n) {
		if (!cnt[v[r]]) {
			cnt[v[r++]]++;
		}
		else {
			ret += (r - l);
			cnt[v[l++]]--;
		}
	}
	ret += (ull)(r - l) * (r - l + 1) / 2;

	cout << ret << "\n";
	return 0;
}
```
