# 23246 Sport Climbing Combined

# 비교 기준이 3개 이상일 때

```c++
#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

vector<pair<int, int>> v1; // mul
map<int,int> v2; // sum
int n;
int a, b, c, d;

bool check(pair<int, int> p1, pair<int, int> p2) {
    if (p1.second == p2.second) {
        if (v2[p1.first] == v2[p2.first]) return p1.first < p2.first;
        return v2[p1.first] < v2[p2.first];
    }
    return p1.second < p2.second;
}

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a >> b >> c >> d;
        int mul = b * c * d;
        int sum = b + c + d;

        v1.push_back({ a,mul });
        v2[a] = sum;
    }

    sort(v1.begin(), v1.end(), check);

    for (int i =0; i < 3;i++)
        cout << v1[i].first << " ";
    cout << "\n";

    return 0;
}
```
