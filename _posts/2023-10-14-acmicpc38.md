# 2170 선 긋기

## array

```c++
#include <iostream>
#include <algorithm>
using namespace std;

typedef pair<int, int> p;
p L[1000004];
int n, f, t, ret, l, r;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> f >> t;
        L[i] = p(f, t);
    }

    sort(L, L + n);

    l = L[0].first; r = L[0].second;

    for (int i = 1; i < n; i++) {
        if (r >= L[i].second) continue;
        else if (r >= L[i].first && r < L[i].second)
            r = L[i].second;
        else {
            ret += r - l;
            l = L[i].first;
            r = L[i].second;
        }
    }
    ret += r - l;
    cout << ret << "\n";
    return 0;
}
```
