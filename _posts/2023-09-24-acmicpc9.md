# 1620 나는야 포켓몬 마스터 이다솜

## map 2개

```c++
#include <iostream>
#include <map>
using namespace std;

int n, m;
string s;
map<string, int> mp1;
map<int, string> mp2;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	cin >> n >> m;

	for (int i = 0; i < n; i++)
	{
		cin >> s;
		mp1[s] = i + 1;
		mp2[i + 1] = s;
	}

	for (int i = 0; i < m; i++)
	{
		cin >> s; 
		if (atoi(s.c_str()))
			cout << mp2[atoi(s.c_str())] << "\n";
		else
			cout << mp1[s] << "\n";
	}
	return 0;
}
//느낀점:
//1. string,int로 찾을 땐 map; int,string으로 찾을 땐 map 또는 array;
```
