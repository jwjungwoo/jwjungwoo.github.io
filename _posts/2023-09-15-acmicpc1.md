# 28445 알록달록 앵무새

## set으로 풀기

```c++
#include <iostream>
#include <set>
using namespace std;
#define xx first
#define yy second
string arr[4];
void solve() {
	for (int i = 0; i < 4; i++)
		cin >> arr[i];
	set<pair<string, string>> s;

	for (int i = 0; i < 4; i++)
		for (int j = 0; j < 4; j++)
			//s.insert(pair<string, string>(arr[i], arr[j]));
			s.emplace(arr[i], arr[j]);

	for (auto it : s)
		cout << it.xx << " " << it.yy << "\n";
}

int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	solve();
	return 0;
}
```

## unique로 풀기

```c++
#include <iostream>
#include <vector>
#include <algorithm>

#define _unique(v) (v).erase(unique((v).begin(),(v).end()),(v).end())
using namespace std;

int main()
{
	cin.tie(NULL); cout.tie(NULL);

	string s;
	vector<string> v;

	int cnt = 4;

	while (cnt--)
	{
		cin >> s;
		v.push_back(s);
	}
	
	sort(v.begin(), v.end());

	_unique(v);

	for (int i = 0; i < v.size(); i++)
		for (int j = 0; j < v.size(); j++)
			cout << v[i] << " " << v[j] << "\n";

	return 0;
}
```
