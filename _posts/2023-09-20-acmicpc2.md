# 2776 암기왕
## binary search
```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int check(int temp, vector<int>& v) {
	int l = 0; int r = v.size() - 1; int mid;
	while (l <= r)
	{
		mid = (l + r) / 2;
		if (v[mid] == temp)
			return 1;
		else if (v[mid] < temp)
			l = mid + 1;
		else
			r = mid - 1;
	}
	return 0;
}

int main() {
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	int t, n, m, num;

	cin >> t;

	for (int i = 0; i < t; i++)
	{
		vector<int> v;
		cin >> n;
		for (int j = 0; j < n; j++)
		{
			cin >> num;
			v.push_back(num);
		}
		sort(v.begin(), v.end());
		cin >> m;
		for (int h = 0; h < m; h++)
		{
			cin >> num;
			cout << check(num, v) << "\n";
		}
	}
	return 0;
}

//느낀점:
//원소 수가 많을 때 함수에서 이용하려면 참조를 이용.
//원소 수가 많을 때 binary search를 이용.
```
