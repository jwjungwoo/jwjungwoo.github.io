# 16282 Black Chain

## math

```c++
#include <iostream>
using namespace std;

typedef unsigned long long ull;
ull ret;
int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    ull n;
    ull i = 1;

    cin >> n;

    while (true) {
        ull sum = i;
        ull temp = i + 1;
        for (ull j = 0; j < i + 1; j++) {
            sum += (temp);
            temp *= 2;
        }
        if (sum >= n) {
            ret = i;
            break;
        }
        i++;
    }

    cout << ret << "\n";
    return 0;
}
```
