# 1012 유기농 배추

## dfs

```c++
#include <iostream>
using namespace std;

const int max_n = 54;
int dy[] = { -1, 0, 1, 0 };
int dx[] = { 0, 1, 0, -1 };
int T, M, N, K, a1, a2, ny, nx, ret;
int a[max_n][max_n];
bool visited[max_n][max_n];

void dfs(int y, int x)
{
	visited[y][x] = 1;
	for (int i = 0; i < 4; i++)
	{
		ny = y + dy[i];
		nx = x + dx[i];
		if (ny < 0 || ny >= N || nx < 0 || nx >= M) continue;
		if (a[ny][nx] && visited[ny][nx] == 0) dfs(ny, nx);
	}
	return;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	cin >> T;
	while (T--)
	{
		cin >> M >> N >> K;
		fill(&a[0][0], &a[0][0] + 54 * 54, 0);
		fill(&visited[0][0], &visited[0][0] + 54 * 54, 0);
		ret = 0;
		while (K--)
		{
			cin >> a1 >> a2;
			a[a2][a1] = 1;
		}

		for (int i = 0; i < N; i++)
		{
			for (int j = 0; j < M; j++)
			{
				if (a[i][j] && !visited[i][j])
				{
					ret++; dfs(i, j);
				}
			}
		}
		cout << ret << "\n";
	}
	return 0;
}
```
