# 2588 곱셈

## (int)(s[0]-'0')

```c++
#include <iostream>
#include <string>
using namespace std;

void solve(int a, int b)
{
	string s = to_string(b);
	cout << a * ((int)(s[2] - '0')) << "\n";
	cout << a * ((int)(s[1] - '0')) << "\n";
	cout << a * ((int)(s[0] - '0')) << "\n";
	cout << a * b << "\n";
}

int main()
{
	int aa, bb;
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	cin >> aa >> bb;

	solve(aa, bb);

	return 0;
}
```
