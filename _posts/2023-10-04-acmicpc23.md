# 2828 사과 담기 게임

## 간단하게 만들기

```c++
#include <iostream>
using namespace std;

int n, m, j, a, ret, l, r;

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	cin >> n >> m;
	cin >> j;
	l = 1;

	while (j--)
	{
		cin >> a;
		r = l + m - 1;
		if (l <= a && a <= r) continue;
		else
		{
			if (a < l)
			{
				ret += l - a;
				l = a; 
			}
			else
			{
				ret += a - r;
				l += a - r; 
			}
		}
	}
	cout << ret << "\n";
	return 0;
}
```
