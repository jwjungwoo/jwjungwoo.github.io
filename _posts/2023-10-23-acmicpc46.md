# 1316 그룹 단어 체커

## prior char

```c++
#include <iostream>
using namespace std;

int n;
string s;
char temp;
int arr[26];
bool flag;
int ret;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n;

	for (int i = 0; i < n; i++) {
		cin >> s;
		flag = 0;
		temp = s[0];
		for (int j = 1; j < s.length(); j++) {
			if (s[j] != temp) {
				if (arr[s[j] - 'a']) {
					flag = 1; break;
				}
				arr[temp - 'a']++;
				if (arr[temp - 'a'] == 2) {
					flag = 1; break;
				}
				temp = s[j];
			}
			else
				temp = s[j];
		}

		if (!flag) ret++;
		fill(&arr[0], &arr[0] + 26, 0);
	}
	cout << ret << "\n";
	return 0;
}
```
