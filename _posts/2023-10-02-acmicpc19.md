# 1157 단어 공부

## array

```c++
#include <iostream>
using namespace std;

string s;

int arr[26];
int ret, flag, whichh;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> s;
    for (int i = 0; i < s.length(); i++) {
        char a = s[i];
        if (a >= 'a' && a <= 'z')
            arr[a - 'a']++;
        else
            arr[a - 'A']++;
    }

    for (int i = 0; i < 26; i++)
    {
        if (ret <= arr[i]) {
            ret = arr[i];
            whichh = i;
        }
    }

    for (int i = 0; i < 26; i++)
        if (arr[i] == ret)
            flag++;

    if (flag >= 2)
        cout << "?\n";
    else
        cout << char('A' + whichh) << "\n";
    return 0;
}
```
