# 30022 행사 준비

## vector에 인덱스와 두 수의 차 저장

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<pair<int, int>> v;

int n, a, b;
int pi, qi;
int arr1[100004];
int arr2[100004];
unsigned long long sum;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n >> a >> b;

    for (int i = 0; i < n; i++)
    {
        cin >> pi >> qi;
        arr1[i + 1] = pi;
        arr2[i + 1] = qi;
        v.push_back({ pi - qi, i + 1 });
    }

    sort(v.begin(), v.end());

    for (int i = 0; i < a; i++)
        sum += arr1[v[i].second];
    for (int i = a; i < v.size(); i++)
        sum += arr2[v[i].second];

    cout << sum;
    return 0;
}
```
