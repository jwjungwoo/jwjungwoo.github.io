# 23247 Ten

## implemetation

```c++
//이 코드가 틀리면 그냥 포기하려했지만 맞았다. ㅎㅎ
//그치만 중꺾마라서 포기하진 않았을거같다.
#include <iostream>
using namespace std;

int m, n, ret;
int arr[304][304];

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> m >> n;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cin >> arr[i][j];
            if (arr[i][j] == 10)
                ret++;
        }
    }

    for (int num = 2; num <= 10; num++) { //한줄 2-10
        if (num > n) break;
        for (int i = 0; i < m; i++) { //m행
            for (int j = 0; j < n - num + 1; j++) { //열
                int sum = 0;
                for (int h = 0; h < num; h++) {
                    sum += arr[i][j + h];
                }
                if (sum == 10)
                    ret++;
            }
        }
    }

    for (int num = 2; num <= 10; num++) { //한줄 2-10
        if (num > m) break;
        for (int i = 0; i < n; i++) { //m행
            for (int j = 0; j < m - num + 1; j++) { //열
                int sum = 0;
                for (int h = 0; h < num; h++) {
                    sum += arr[j + h][i];
                }
                if (sum == 10)
                    ret++;
            }
        }
    }

    for (int num = 2; num <= 5; num++) { //2,3,4,5
        if (num == 2)
            continue;
        if (num > n || m < 2) break;
        for (int i = 0; i < m - 1; i++) { //두 줄
            for (int j = 0; j < n - num + 1; j++) {
                int sum = 0;
                for (int h = 0; h < num; h++) {
                    sum += arr[i][j + h];
                    sum += arr[i + 1][j + h];
                }
                if (sum == 10)
                    ret++;
            }
        }
    }

    for (int num = 2; num <= 5; num++) { //2,3,4,5
        if (num > m || n < 2) break;
        for (int i = 0; i < n - 1; i++) { //두 줄
            for (int j = 0; j < m - num + 1; j++) {
                int sum = 0;
                for (int h = 0; h < num; h++) {
                    sum += arr[j + h][i];
                    sum += arr[j + h][i + 1];
                }
                if (sum == 10)
                    ret++;
            }
        }
    }

    if (n >= 3 && m >= 3) {
        for (int i = 0; i < m - 3 + 1; i++) {
            for (int j = 0; j < n - 3 + 1; j++) {
                int sum = 0;
                for (int h = 0; h < 3; h++)
                    for (int k = 0; k < 3; k++)
                        sum += arr[i + h][j + k];
                if (sum == 10)
                    ret++;
            }
        }
    }
    cout << ret << "\n";
    return 0;
}
```
