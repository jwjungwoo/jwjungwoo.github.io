# 15686 치킨 배달

## simply

```c++
#include <iostream>
#include <vector>
using namespace std;

const int max_n = 54;
int arr[max_n][max_n];

int n, m;
int ret = 999999999;

vector<pair<int, int>> h, c;
vector<vector<int>> clist;
vector<int> v;

void combi(int start, vector<int> v) {
    if (v.size() == m) {
        clist.push_back(v);
        return;
    }
    for (int i = start + 1; i < c.size(); i++) {
        v.push_back(i);
        combi(i, v);
        v.pop_back();
    }
    return;
}

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> arr[i][j];
            if (arr[i][j] == 1) h.push_back({ i,j });
            if (arr[i][j] == 2) c.push_back({ i,j });
        }
    }

    combi(-1, v);
    
    for (vector<int> cList : clist) {
        int _ret = 0;
        for (pair<int, int> home : h) {
            int _min = 987654321;
            for (int i : cList) {
                int a = abs(home.first - c[i].first) + abs(home.second - c[i].second);
                _min = min(_min, a);
            }
            _ret += _min;
        }
        ret = min(_ret, ret);
    }
    cout << ret << "\n";
    return 0;
}
```
