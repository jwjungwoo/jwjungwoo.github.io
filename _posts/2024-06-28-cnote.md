## pq
```c
백준 2109 번은 pq 로 구현해야 시간초과가 안 난다.
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>
#define MAX_SIZE 10004

void swap(int* a, int* b) {
	int temp = *b;
	*b = *a;
	*a = temp;
}

typedef struct {
	int heap[MAX_SIZE];
	int count;
}priorityQueue;

void push(priorityQueue *pq, int data) {
	if (pq->count >= MAX_SIZE) return;
	pq->heap[pq->count] = data;
	int now = pq->count;
	int parent = (now - 1) / 2;

	while (now > 0 && pq->heap[now] > pq->heap[parent]) {
		swap(&pq->heap[now], &pq->heap[parent]);
		now = parent;
		parent = (parent - 1) / 2;
	}
	pq->count++;
}

int pop(priorityQueue* pq) {
	if (pq->count <= 0) return -9999;
	int res = pq->heap[0];
	pq->count--;
	pq->heap[0] = pq->heap[pq->count];
	int now = 0;
	int leftChild = 1;
	int rightChild = 2;
	int target = now;

	while (leftChild < pq->count) {
		if (pq->heap[target] < pq->heap[leftChild]) {
			target = leftChild;
		}
		if (pq->heap[target] < pq->heap[rightChild] && rightChild < pq->count) {
			target = rightChild;
		}
		if (target == now) {
			break;
		}
		else {
			swap(&pq->heap[now], &pq->heap[target]);
			now = target;
			leftChild = now * 2 + 1;
			rightChild = now * 2 + 2;
		}
	}
	return res;
}

int main() {
	int n, data;
	scanf("%d", &n);
	priorityQueue pq;
	pq.count = 0;
	for (int i = 0; i < n;i++) {
		scanf("%d", &data);
		push(&pq, data);
	}

	for (int i = 0; i < n; i++) {
		int data = pop(&pq);
		printf("%d ", data);
	}
	return 0;
}
```

## map
```c
#include <stdio.h>
#include <string.h>

#define MAX_NODES 100  // 트리 최대 크기

typedef struct {
    char key[20];  // 키값 (문자열)
    int value;     // 저장할 데이터
    int used;      // 해당 노드 사용 여부
} Node;

Node tree[MAX_NODES];  // 정적 배열 기반 트리

// 삽입 함수
void insert(const char* key, int value) {
    int index = 0;

    while (index < MAX_NODES) {
        if (!tree[index].used) {  // 비어있는 곳에 삽입
            strcpy(tree[index].key, key);
            tree[index].value = value;
            tree[index].used = 1;
            return;
        }

        // BST 규칙에 따라 왼쪽 또는 오른쪽으로 이동
        if (strcmp(key, tree[index].key) < 0)
            index = 2 * index + 1;  // 왼쪽 자식
        else
            index = 2 * index + 2;  // 오른쪽 자식
    }
}

// 검색 함수
int search(const char* key) {
    int index = 0;

    while (index < MAX_NODES && tree[index].used) {
        int cmp = strcmp(key, tree[index].key);
        if (cmp == 0)
            return tree[index].value;
        else if (cmp < 0)
            index = 2 * index + 1;  // 왼쪽 이동
        else
            index = 2 * index + 2;  // 오른쪽 이동
    }
    return -1;  // 찾지 못한 경우
}

// 중위 순회 (정렬된 출력)
void inorder(int index) {
    if (index >= MAX_NODES || !tree[index].used) return;
    inorder(2 * index + 1);  // 왼쪽 자식
    printf("%s: %d\n", tree[index].key, tree[index].value);
    inorder(2 * index + 2);  // 오른쪽 자식
}

// 테스트
int main() {
    insert("banana", 200);
    insert("apple", 100);
    insert("grape", 300);

    printf("검색 결과: apple = %d\n", search("apple"));
    printf("검색 결과: banana = %d\n", search("banana"));
    printf("검색 결과: grape = %d\n", search("grape"));
    printf("검색 결과: orange = %d\n", search("orange"));

    printf("\n정렬된 출력:\n");
    inorder(0);

    return 0;
}
```
