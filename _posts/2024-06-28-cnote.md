# math.h

## sqrt

```c
#include <stdio.h>
#include <math.h>

void main() {
	float x1, x2, a, b, c;
	scanf("%f %f %f", &a, &b, &c);
	x1 = (-b + sqrt(b * b - 4 * a * c)) / 2. / a;
	x2 = (-b - sqrt(b * b - 4 * a * c)) / 2. / a;
	printf("%f %f", x1, x2);
}
```
