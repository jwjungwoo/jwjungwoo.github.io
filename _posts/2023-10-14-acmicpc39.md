# 3273 두 수의 합

## while 조건 시 l <= r하면 안 됨

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int n, x, ret, l, r;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;
    vector<int> v(n);
    for (int i = 0; i < n; i++) cin >> v[i];
    cin >> x;

    sort(v.begin(), v.end());

    l = 0; r = n - 1;

    while (l < r) { \\ l <= r은 l == r 이 되어 같은 정수가 2번 뽑힐 수도 있기에 안 됨.
        if (v[l] + v[r] == x)
            ret++, l++;
        else if (v[l] + v[r] > x)
            r--;
        else
            l++;
    }
    cout << ret << "\n";
    return 0;
}
```
