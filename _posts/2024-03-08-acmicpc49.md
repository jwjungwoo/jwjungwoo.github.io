# 2745 진법 변환

## 구현

```c++
#include <iostream>
#include <cmath>
using namespace std;

string n, k;
int b;
int ret;

int main() {
	
	cin >> n >> b;
	reverse(n.begin(), n.end());
	int l = n.length();

	for (int i = 0; i < l; i++) {
		if (n[i] >= '0' && n[i] <= '9') {
			ret += int(n[i] - '0') * pow(b,i);
		}
		else {
			ret += (int(n[i]) - int('A') + 10)*pow(b,i);
		}
	}

	cout << ret << "\n";
	return 0;
}
```
