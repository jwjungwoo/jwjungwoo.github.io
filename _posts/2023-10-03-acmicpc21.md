# 2583 영역 구하기

## dfs

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int max_n = 140;
int dy[] = { 0,-1,0,1 };
int dx[] = { 1,0,-1,0 };
int a[max_n][max_n], visited[max_n][max_n];
int m, n, k, ny, nx, cnt;
vector<int> v;

int dfs(int y, int x)
{
	int ret = 1;
	visited[y][x] = 1;
	for (int i = 0; i < 4; i++)
	{
		ny = y + dy[i];
		nx = x + dx[i];
		if (ny < 0 || ny >= m || nx < 0 || nx >= n) continue;
		if (!a[ny][nx] || visited[ny][nx]) continue;
		ret += dfs(ny, nx);
	}
	return ret;
}


int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	cin >> m >> n >> k;

	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			a[i][j] = 1;
		}
	}

	while (k--)
	{
		int x1, y1, x2, y2;
		cin >> x1 >> y1 >> x2 >> y2;

		for (int i = y1; i < y2; i++)
		{
			for (int j = x1; j < x2; j++)
			{
				a[i][j] = 0;
			}
		}
	}

	cnt = 0;
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (a[i][j] && !visited[i][j])
			{
				cnt++; v.push_back(dfs(i, j));
			}
		}
	}
	
	sort(v.begin(), v.end());
	cout << cnt << "\n";
	for (int i : v) cout << i << " ";
	return 0;
}
```
