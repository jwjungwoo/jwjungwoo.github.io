# 3652 새트리

## recursion

```c++
#include <iostream>
using namespace std;

typedef long long ll;
string y;

string go(ll a, ll b, string s) {
    if (a == b)
        return s;
    if (a > b) {
        s += 'R';
        return go(b, a - b, s);
    }
    else {
        s += 'L';
        return go(b - a, a, s);
    }
}

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> y;
    int idx = y.find("/");

    cout << go(atoi(y.substr(0,idx).c_str()), atoi(y.substr(idx + 1,y.length() - idx -1).c_str()), "");
    return 0;
}
```
