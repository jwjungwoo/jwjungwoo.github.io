# 30021 순열 선물하기

## 홀홀짝짝홀홀짝짝

```c++
#include <iostream>
using namespace std;

int arr[5004];
int n;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	for (int i = 1; i <= 5000; i++)
		arr[i] = i;
	
	cin >> n;

	if (n == 1)
		cout << "YES\n" << 1 << "\n";

	else if (n == 2)
		cout << "NO\n";

	else {
		cout << "YES\n";
		int k = n / 4;

		for (int i = 0; i <= n; i += 4)
		{

			if (i + 1 <= n)
			cout << arr[i + 1] << " ";

			if (i + 3 <= n)
				cout << arr[i + 3] << " ";

			if (i + 2 <= n)
				cout << arr[i + 2] << " ";

			if (i + 4 <= n)
				cout << arr[i + 4] << " ";
		}
	}

	return 0;
}
```
