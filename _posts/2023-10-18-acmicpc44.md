# 1700 멀티탭 스케줄링

## memory paging optimal 1

```c++
#include<bits/stdc++.h>
using namespace std;
const int INF = 987654321;
int n, k;
vector<int> v;

int arr[104];
int visited[104];
int ret;
int main() {
	
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n >> k;
	for (int i = 0; i < k; i++) cin >> arr[i];
	for (int i = 0; i < k; i++) {
		if (!visited[arr[i]]) {
			if (v.size() == n) {
				int last_idx = 0, pos;
				for (int _a : v) {
					int here_pick = INF;
					for (int j = i + 1; j < k; j++) {
						if (arr[j] == _a) {
							here_pick = j; break;
						}
					}
					if (last_idx < here_pick) {
						last_idx = here_pick;
						pos = _a;
					}
				}
				visited[pos] = 0;
				ret++;
				v.erase(find(v.begin(), v.end(), pos));
			}
			v.push_back(arr[i]); visited[arr[i]] = 1;
		}
	}
	cout << ret << "\n";
	return 0;
}
```

## memory paging optimal 2

```c++
#include <iostream>
#include <vector>
using namespace std;

int n, k;
int cnt[104];
vector<int> v;

int arr[104];
int temp;
int ret;
bool flag;
int main() {
	
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n >> k;
	for (int i = 0; i < k; i++) { 
		cin >> arr[i];
		if (cnt[arr[i]] == 0)
			cnt[arr[i]] = i;
	}

	//main
	for (int i = 0; i < k; i++) {

		if (v.size() == n) {
			flag = false;
			for (int j = 0; j < n; j++)
				if (v[j] == arr[i]) {
					flag = true;
					break;
				}

			if (flag) {
				cnt[arr[i]] = 0;
				for (int j = i + 1; j < k; j++) {
					if (arr[j] == arr[i]) {
						cnt[arr[i]] = j;
						break;
					}
				}
				continue;
			}

			//뽑을 놈

			int max_n = 0;
			int r = 0;
			int t = 0; 
			for (int j = 0; j < n; j++) { //뽑을 숫자 r
				if (cnt[v[j]] == 0) {
					r = v[j];
					t = j;
					break;
				}
				if (max_n < cnt[v[j]]) {
					r = v[j];
					t = j;
					max_n = cnt[v[j]];
				}
			}
			//지움
			v.erase(v.begin() + t, v.begin() + t + 1);
			//입력
			v.push_back(arr[i]);
			cnt[arr[i]] = 0;
			for (int j = i + 1; j < k; j++) {
				if (arr[j] == arr[i]) {
					cnt[arr[i]] = j;
					break;
				}
			}
			ret++;
		}
		// v.size() != n
		else {
			flag = false;
			for (int j = 0; j < v.size(); j++)
				if (arr[i] == v[j]) { flag = true; break; }
			if (flag) {
				cnt[arr[i]] = 0;
				for (int j = i + 1; j < k; j++) {
					if (arr[j] == arr[i]) {
						cnt[arr[i]] = j;
						break;
					}
				}
				continue;
			}
			v.push_back(arr[i]);
			cnt[arr[i]] = 0;
			for (int j = i + 1; j < k; j++) {
				if (arr[j] == arr[i]) {
					cnt[arr[i]] = j;
					break;
				}
			}
		}
	}

	cout << ret << "\n";
	return 0;
}
```
