# 3986 좋은 단어

## stack

```c++
#include <iostream>
#include <stack>
using namespace std;

int cnt, n;
string s;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> n;
	while (n--) {
		stack<char> stk;
		cin >> s;
		for (int i = 0; i < s.length(); i++)
		{
			if (stk.size() == 0 || stk.top() != s[i])
				stk.push(s[i]);
			else stk.pop();
		}
		if (!stk.size())
			cnt++;
	}
	cout << cnt << "\n";
	return 0;
}
```
