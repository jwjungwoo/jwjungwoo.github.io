# scanf, printf

```c
//double은 scanf에선 무조건 lf 써야하나 printf에선 lf, f 둘 다 됨
#include <stdio.h>
int main(){ 
 int i ;
 char c ;
 float f ;
 double d ; 
 scanf("%d %c %f %lf", &i, &c, &f, &d) ;
 printf("%d %c %f %f\n", i, c, f, d) ;
}
```
