# 20040 사이클 게임

## 결합시 높이 2인 트리로 만들기

```c++
#include <iostream>
#include <map>
using namespace std;

int n, m;
int c, a, b;
map<int, int> root;

int find_root(int x) {
    if (x == root[x]) return x;
    return root[x] = find_root(root[x]);
}

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    cin >> n >> m;

    for (int i = 0; i <= n; i++)
        root[i] = i;

    for (int i = 0; i < m; i++) {
        cin >> c >> a >> b;
        if (c == 0) {
            int aa = find_root(a);
            int bb = find_root(b);

            if (aa != bb) {
                root[bb] = aa;
            }
        }

        else {
            if (find_root(a) == find_root(b)) cout << "YES\n";
            else cout << "NO\n";
        }
    }

    return 0;
}
```
