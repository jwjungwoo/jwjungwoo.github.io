# 1781 컵라면

## priority queue

```c++
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
using namespace std;

int n, a, b;
typedef unsigned long long ll;
ll ret;
priority_queue < ll, vector<ll>, greater<ll>> pq;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;

    vector<pair<ll, ll>> v(n);

    for (int i = 0; i < n; i++)
        cin >> v[i].first >> v[i].second;

    sort(v.begin(), v.end());

    for (int i = 0; i < n; i++) {
        pq.push(v[i].second);
        if (pq.size() > v[i].first) {
            pq.pop();
        }
    }

    while (pq.size()) {
        ret += pq.top(); pq.pop();
    }
    cout << ret << "\n";
    return 0;
}
```
