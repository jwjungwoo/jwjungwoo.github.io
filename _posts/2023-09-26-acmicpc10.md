# 1213 팰린드롬 만들기

## ret에 더하고 reverse

```c++
#include <iostream>
#include <algorithm>
using namespace std;

int arr[30]; // A는 0, B는 1, ..., 배열크기는 넉넉잡아 +4
string input, ret, temp;
int flag, c;

int main() {

	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

	cin >> input;

	for (int i = 0; i < int(input.length()); i++)
		arr[input[i] - 'A']++;
	for (int i = 0; i < 26; i++)
	{
		if (arr[i] % 2)
		{
			flag++;
			c = i;
		}
	}

	if (flag >= 2)
		cout << "I'm Sorry Hansoo\n";
	else {
		for (int i = 0; i <26; i++)
			for (int j = 0; j < arr[i]/2; j++)
				ret += char('A' + i);

		temp = ret;
		reverse(ret.begin(),ret.end());
		if (flag == 1)
			temp += char('A' + c);
		temp += ret;

		cout << temp;
	}
	return 0;
}
```
