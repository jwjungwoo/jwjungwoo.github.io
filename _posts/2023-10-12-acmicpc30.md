# 25945 컨테이너 재배치

## math

```c++
#include <iostream>
#include <vector>
using namespace std;

int n, m, num, sum, avg;
vector<int> v;
int ret;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> num;
        v.push_back(num);
        sum += num;
    }

    avg = sum / n; 
    int k = 0;      //평균보다 작은 팀 컨베이너 수 차이
    int kk = 0;     //평균 + 1 보다 높은 팀 컨베이너 수 차이

    for (int i = 0; i < n; i++) {
        if (v[i] < avg)
            k += (avg - v[i]);
        else if (v[i] > avg + 1)
            kk += (v[i] - avg - 1);
    }

    cout << max(k,kk) << "\n";      //작 vs 높의 갯수 중 누가 더 차이가 큰가
    return 0;
}
```
