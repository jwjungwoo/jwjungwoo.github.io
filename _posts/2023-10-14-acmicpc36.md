# 1202 보석 도둑

## priority queue

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

int n, k, m, v;
int c;
typedef unsigned long long ull;
ull ret;

int main() {

    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    cin >> n >> k;

    vector<pair<ull, ull>> v1(n);
    vector<ull> v2(k);

    for (int i = 0; i < n; i++)
        cin >> v1[i].first >> v1[i].second;

    for (int i = 0; i < k; i++)
        cin >> v2[i];
    sort(v1.begin(), v1.end());
    sort(v2.begin(), v2.end());
    priority_queue<ull> pq;

    int j = 0;
    for (int i = 0; i < k; i++) {
        while (j < n && v2[i] >= v1[j].first) pq.push(v1[j++].second);
        if (pq.size()) {
            ret += pq.top(); pq.pop();
        }
    }

    cout << ret << "\n";
    return 0;
}
```
